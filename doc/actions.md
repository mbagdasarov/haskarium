# Действия и процедуры

```haskell
{- действие (процедура) `j`
   с эффектом, описываемым типом `k`,
   и результатом типа `m` -}
j :: k m
j = do  -- последовательное выполнение действий
    действие...
    действие...

main :: IO ()
main = putStrLn "Hello World"

---

{- действие (процедура) `n`

   с эффектом, описываемым типом `r`,
   и результатом типа `s` -}
n :: o -> p -> q -> r s
n o p q = do
    действие...
    действие...

```
